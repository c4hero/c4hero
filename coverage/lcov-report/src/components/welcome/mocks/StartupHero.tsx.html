import { FolderOpen, FilePlus2, FileText, ChevronRight, Github } from 'lucide-react'
import type { RecentFolder } from './PreviewStartup'
import { formatRelative } from './formatRelative'
 
interface Props {
  recents: RecentFolder[]
  onCreateCollection: () => void
  onOpenCollection: () => void
  onOpenRecent: (name: string) => void
  onOpenFile: () => void
}
 
/**
 * Option B — "Hero"
 *
 * Pattern: a marketing-grade landing layout. Left half is dominated by a
 * stylised C4-diagram preview that signals "this is what you'll be making".
 * Right half holds the actions in a tight column. The diagram shows
 * three nodes with labelled edges and a subtle gradient backdrop — high
 * impression, low cognitive load. On mobile it becomes a top-stacked
 * strip with a smaller preview and the actions below it.
 */
export default function StartupHero({
  recents,
  onCreateCollection,
  onOpenCollection,
  onOpenRecent,
  onOpenFile,
}: Props) {
  return (
    <main
      style={{
        flex: 1,
        display: 'grid',
        gridTemplateColumns: 'minmax(0, 1fr) minmax(320px, 460px)',
        gap: 32,
        padding: '40px 24px',
        maxWidth: 1280,
        width: '100%',
        margin: '0 auto',
        alignItems: 'center',
      }}
      // Auto-stack on small viewports without media queries
      ref={(el) => {
        if (!el) return
        const apply = () => {
          el.style.gridTemplateColumns = window.innerWidth < 880
            ? 'minmax(0, 1fr)'
            : 'minmax(0, 1fr) minmax(320px, 460px)'
        }
        apply()
        // (one-shot apply is fine; resize handlers would attach more state than this preview needs)
      }}
    >
      {/* Hero illustration */}
      <div
        style={{
          position: 'relative',
          aspectRatio: '4 / 3',
          maxHeight: 480,
          borderRadius: 22,
          background: 'radial-gradient(circle at 25% 25%, color-mix(in srgb, var(--color-accent) 18%, transparent) 0%, transparent 55%), radial-gradient(circle at 80% 75%, color-mix(in srgb, #a78bfa 14%, transparent) 0%, transparent 60%), var(--color-surface-1)',
          border: '1px solid var(--color-border)',
          overflow: 'hidden',
        }}
      >
        <DiagramPreview />
        {/* Caption */}
        <div
          style={{
            position: 'absolute',
            left: 18,
            bottom: 16,
            display: 'inline-flex',
            alignItems: 'center',
            gap: 8,
            padding: '6px 10px',
            borderRadius: 8,
            background: 'color-mix(in srgb, var(--color-bg-primary) 75%, transparent)',
            backdropFilter: 'blur(8px)',
            border: '1px solid var(--color-border)',
            fontSize: 11,
            fontWeight: 500,
            color: 'var(--color-text-secondary)',
          }}
        >
          <span style={{ width: 6, height: 6, borderRadius: 99, background: '#22c55e' }} />
          Live preview · Container view
        </div>
      </div>
 
      {/* Action column */}
      <div style={{ display: 'flex', flexDirection: 'column', gap: 22 }}>
        <div style={{ display: 'flex', flexDirection: 'column', gap: 10 }}>
          <Logo />
          <h1 style={{ margin: 0, fontSize: 30, fontWeight: 700, letterSpacing: '-0.025em', lineHeight: 1.1, color: 'var(--color-text-primary)' }}>
            Architecture diagrams that{' '}
            <span style={{ color: 'var(--color-accent)' }}>live with your code.</span>
          </h1>
          <p style={{ margin: 0, fontSize: 14, lineHeight: 1.55, color: 'var(--color-text-muted)', maxWidth: 380 }}>
            Visual canvas, C4 model, plain-text <code style={{ fontSize: 13 }}>.dsl</code> files. No accounts. No cloud. Just folders on your machine.
          </p>
        </div>
 
        {/* Primary actions, stacked */}
        <div style={{ display: 'flex', flexDirection: 'column', gap: 8 }}>
          <PrimaryCTA
            icon={<FilePlus2 size={16} />}
            label="New collection"
            sublabel="Create a folder for your workspaces"
            onClick={onCreateCollection}
          />
          <SecondaryCTA
            icon={<FolderOpen size={16} />}
            label="Open existing collection"
            onClick={onOpenCollection}
          />
        </div>
 
        {/* Recents — compact chips */}
        {recents.length > 0 ? (
          <section style={{ display: 'flex', flexDirection: 'column', gap: 6 }}>
            <span style={{ fontSize: 10, fontWeight: 600, letterSpacing: '0.08em', textTransform: 'uppercase', color: 'var(--color-text-muted)' }}>
              Continue
            </span>
            <div style={{ display: 'flex', gap: 6, flexWrap: 'wrap' }}>
              {recents.slice(0, 4).map((r) => (
                <button
                  key={r.name}
                  onClick={() => onOpenRecent(r.name)}
                  style={{
                    display: 'inline-flex',
                    alignItems: 'center',
                    gap: 6,
                    padding: '5px 10px',
                    fontSize: 12,
                    borderRadius: 99,
                    background: 'var(--color-surface-2)',
                    border: '1px solid var(--color-border)',
                    color: 'var(--color-text-secondary)',
                    cursor: 'pointer',
                  }}
                  onMouseEnter={(e) => { e.currentTarget.style.borderColor = 'var(--color-accent)'; e.currentTarget.style.color = 'var(--color-accent)' }}
                  onMouseLeave={(e) => { e.currentTarget.style.borderColor = 'var(--color-border)'; e.currentTarget.style.color = 'var(--color-text-secondary)' }}
                  title={`${formatRelative(r.modifiedAt)}`}
                >
                  <FolderOpen size={11} />
                  {r.displayName ?? r.name}
                </button>
              ))}
            </div>
          </section>
        ) : null}
 
        <div style={{ display: 'flex', alignItems: 'center', gap: 14, flexWrap: 'wrap', fontSize: 11, color: 'var(--color-text-muted)' }}>
          <button
            onClick={onOpenFile}
            style={{ display: 'inline-flex', alignItems: 'center', gap: 4, background: 'transparent', border: 'none', color: 'var(--color-text-muted)', fontSize: 11, cursor: 'pointer', padding: 0 }}
          >
            <FileText size={11} />
            Open .dsl file
          </button>
          <span>·</span>
          <a
            href="#"
            onClick={(e) => e.preventDefault()}
            style={{ display: 'inline-flex', alignItems: 'center', gap: 4, color: 'var(--color-text-muted)', textDecoration: 'none', fontSize: 11 }}
          >
            <Github size={11} />
            View on GitHub
          </a>
        </div>
      </div>
    </main>
  )
}

}
